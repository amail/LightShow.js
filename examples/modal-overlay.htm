Modal overlay example

This example demonstrates the flexibility of LightShow and how it can be used to create modals.

Markup 
<div id="document-overlay-content" class="overlay-modal">
	<div class="header">
		<h1>Sign up and be awesome!</h1>
		<a href='#' class='hide-button' title='Close sign up'>X</a>
	</div>
	<div class="content">
		<ul class="inputs">
			<li>
				<label>
					<h3>Username</h3>
					<input type="text" name="username" placeholder="e.g. mrmiagi023" />
				</label>
			</li>
			<li>
				<label>
					<h3>Email Address</h3>
					<input type="text" name="email_address" placeholder="e.g. jeff@peffpoa.com" />
				</label>
			</li>
			<li>
				<label>
					<h3>Password</h3>
					<input type="password" name="password" />
				</label>
			</li>
		</ul>
		<input type="button" value="Sign up" />
	</div>
</div>
Code 
var signup_modal = $.lightShow({
	content: $("div#document-overlay-content"),
	showOnStartup: true,
	effect: {
		hideContent: function(element, completion){
			element.animate({'opacity':0, 'top':'-=50'}, 700).hide(function(){
				$(this).css('opacity', 1);
				signup_modal.readapt();
				completion();
			});
		}
	}
});

$("div#document-overlay-content input[type='button']").click(function(){
	signup_modal.hide();
	return false;
});
